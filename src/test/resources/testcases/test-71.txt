    allow jcr:read for "Test Group",user1
end
set ACL on /content
    allow jcr:read for "Test Group- Cool People","Test Group",user1
end

    allow jcr:read on /content
end

    remove * on /libs,/apps

end

    remove * for user1,"Test Group",user2
end

end
